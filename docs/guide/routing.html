Routing

Define routes and middleware to handle HTTP requests and direct them to the appropriate controllers.

Basic Routing

Simple Routes

Define routes in routes/web.php:

<?php

use App\Core\Route;
use App\Controllers\HomeController;
use App\Controllers\UserController;

// Basic GET route
Route::get('/', [HomeController::class, 'index']);

// Route with parameter
Route::get('/users/{id}', [UserController::class, 'show']);

// Multiple HTTP methods
Route::post('/users', [UserController::class, 'store']);
Route::put('/users/{id}', [UserController::class, 'update']);
Route::delete('/users/{id}', [UserController::class, 'destroy']);

Route Parameters

// Required parameters
Route::get('/users/{id}', [UserController::class, 'show']);

// Optional parameters
Route::get('/posts/{id?}', [PostController::class, 'show']);

// Multiple parameters
Route::get('/users/{userId}/posts/{postId}', [PostController::class, 'show']);

Route Groups

Prefix Groups

// API routes with prefix
Route::group(['prefix' => 'api'], function () {
    Route::get('/users', [ApiController::class, 'users']);
    Route::get('/posts', [ApiController::class, 'posts']);
});

// Admin routes
Route::group(['prefix' => 'admin'], function () {
    Route::get('/dashboard', [AdminController::class, 'dashboard']);
    Route::get('/users', [AdminController::class, 'users']);
});

Middleware Groups

// Protected routes
Route::group(['middleware' => ['auth']], function () {
    Route::get('/profile', [UserController::class, 'profile']);
    Route::post('/logout', [AuthController::class, 'logout']);
});

// Admin only routes
Route::group(['middleware' => ['auth', 'admin']], function () {
    Route::get('/admin/dashboard', [AdminController::class, 'dashboard']);
    Route::resource('/admin/users', AdminUserController::class);
});

Route Middleware

Applying Middleware

// Single middleware
Route::get('/profile', [UserController::class, 'profile'])
     ->middleware('auth');

// Multiple middleware
Route::get('/admin', [AdminController::class, 'index'])
     ->middleware(['auth', 'admin']);

// Middleware with parameters
Route::get('/api/data', [ApiController::class, 'data'])
     ->middleware('throttle:60,1');

Creating Custom Middleware

<?php

namespace App\Middleware;

class CustomMiddleware
{
    public function handle($next)
    {
        // Before request processing
        if (!$this->checkCondition()) {
            http_response_code(403);
            echo json_encode(['error' => 'Access denied']);
            exit;
        }
        
        // Continue to next middleware/controller
        $response = $next();
        
        // After request processing
        // Modify response if needed
        
        return $response;
    }
    
    private function checkCondition()
    {
        // Your middleware logic here
        return true;
    }
}

Resource Routes

RESTful Resources

// Creates all RESTful routes
Route::resource('posts', PostController::class);

// Generated routes:
// GET    /posts           - index()
// GET    /posts/create    - create()  
// POST   /posts           - store()
// GET    /posts/{id}      - show()
// GET    /posts/{id}/edit - edit()
// PUT    /posts/{id}      - update()
// DELETE /posts/{id}      - destroy()

Partial Resources

// Only specific methods
Route::resource('photos', PhotoController::class)
     ->only(['index', 'show']);

// Exclude specific methods
Route::resource('photos', PhotoController::class)
     ->except(['create', 'edit']);

Route Caching

Performance Optimization

For production applications, routes can be cached for better performance:

// Generate route cache
php artisan route:cache

// Clear route cache
php artisan route:clear

Best Practices

1. Organize Routes Logically

// Group related routes
Route::group(['prefix' => 'api/v1', 'middleware' => 'api'], function () {
    // User routes
    Route::resource('users', UserController::class);
    
    // Post routes  
    Route::resource('posts', PostController::class);
    
    // Auth routes
    Route::post('login', [AuthController::class, 'login']);
    Route::post('logout', [AuthController::class, 'logout']);
});

2. Use Descriptive Route Names

Route::get('/users/{id}', [UserController::class, 'show'])
     ->name('users.show');

Route::post('/users', [UserController::class, 'store'])
     ->name('users.store');

3. Validate Route Parameters

// Numeric ID validation
Route::get('/users/{id}', [UserController::class, 'show'])
     ->where('id', '[0-9]+');

// Slug validation
Route::get('/posts/{slug}', [PostController::class, 'show'])
     ->where('slug', '[a-z0-9-]+');

This routing system provides a flexible, powerful way to organize your application's URL structure and handle HTTP requests efficiently.
